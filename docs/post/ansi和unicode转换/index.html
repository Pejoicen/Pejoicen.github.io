{

	p1=wStr.GetBuffer(wStr.GetLength());
	wStr.ReleaseBuffer();

	// Convert to a char*
	size_t s1 = wcslen(p1) + 1;
	const size_t newsize1 = 2000;
	size_t convertedChars1 = 0;
	//char c1[newsize1];
	//wcstombs_s(&convertedChars1, c1, s1, p1 , _TRUNCATE);
	wcstombs_s(&convertedChars1, Char, s1, p1 , _TRUNCATE);
}
