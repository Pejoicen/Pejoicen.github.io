// 在窗口类中增加CToolTipCtrl成员变量
class CDemodulateUnitDlg : public CDialog
{

public:
    CToolTipCtrl m_ToolTip;                 // CToolTipCtrl
}
    
    
// 类成员函数中，重写 PreTranslateMessage
BOOL CDemodulateUnitDlg::PreTranslateMessage(MSG* pMsg)
{
    // TODO: 在此添加专用代码和/或调用基类


    m_ToolTip.RelayEvent(pMsg);

    return CDialog::PreTranslateMessage(pMsg);
}


// 类成员函数中，重写 OnInitDialog
BOOL CDemodulateUnitDlg::OnInitDialog()
{
    CDialog::OnInitDialog();

    // TODO:  在此添加额外的初始化

    EnableToolTips(TRUE);                       // added
    m_ToolTip.Create(this);                     // added
    m_ToolTip.Activate(TRUE);                   // added

    CWnd* pW = GetDlgItem(IDC_DEMO_CARRFREQ);   // added 得到窗口指针
    m_ToolTip.AddTool(pW, "载波频率范围XX~XX"); // added 添加显示的提示信息

    pW = GetDlgItem(IDC_DOWNRF_FREQ);           // added 得到窗口指针
    m_ToolTip.AddTool(pW, "下行频率范围XX~XX"); // added 添加显示的提示信息

    return TRUE;  // return TRUE unless you set the focus to a control
                  // 异常: OCX 属性页应返回 FALSE
}
